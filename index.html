Ejemplo de Consulta de Negocio
Facturas pendientes de pago de un cliente — útil para gestión de morosidad y cobranza.
-- Facturas pendientes de un cliente específico
SELECT Factura.Nro_Factura,
       Factura.Monto_Total,
       Factura.Fecha_Vencimiento
FROM Factura
JOIN Contrato ON Factura.ID_Contrato = Contrato.ID_Contrato
WHERE Contrato.DNI_Cliente = '12345678'
  AND Factura.Estado_Pago = 'Pendiente';
